20 struct VColor

24 constexpr VColor(uint32_t rgba) : rgba(rgba) {}
25 constexpr VColor(uint8_t r = 0, uint8_t g = 0, uint8_t b = 0, uint8_t a = 255)
26 : rgba(uint32_t(r) | (uint32_t(g) << 8) | (uint32_t(b) << 16) | (uint32_t(a) << 24))
27 {
28 }
29
30 constexpr VColor(const char *hex)
31 {
32 rgba = 0;
33 uint32_t shift = 0u;
34 for (int i = 0; i < 9; ++i)
35 {
36 const char c = hex[i];
37 switch (c)
38 {
39 case '0' ... '9':
40 rgba |= (c - '0') << shift;
41 shift += 4u;
42 break;
43 case 'A' ... 'F':
44 rgba |= (c - 'A' + 10) << shift;
45 shift += 4u;
46 break;
47 case '#':
48 break;
49 default:
50 case 0:
51 return;
52 }
53 }
54 }
55
56 constexpr uint8_t getR() const { return (rgba >> 0) & 0xFF; }
57 constexpr uint8_t getG() const { return (rgba >> 8) & 0xFF; }
58 constexpr uint8_t getB() const { return (rgba >> 16) & 0xFF; }
59 constexpr uint8_t getA() const { return (rgba >> 24) & 0xFF; }
60
61 constexpr void setR(uint8_t value) { *this = VColor(value, getG(), getB(), getA()); }
62 constexpr void setG(uint8_t value) { *this = VColor(getR(), value, getB(), getA()); }
63 constexpr void setB(uint8_t value) { *this = VColor(getR(), getG(), value, getA()); }
64 constexpr void setA(uint8_t value) { *this = VColor(getR(), getG(), getB(), value); }
65
66 VE::Math::VVector3 toFloat3() const
67 {
68 return VE::Math::VVector3(((rgba >> 0) & 0xFF) / 255.0f, ((rgba >> 8) & 0xFF) / 255.0f, ((rgba >> 16) & 0xFF) / 255.0f);
69 }
70

75 }

87 constexpr const char_return<9> to_hex() const
88 {
89 char_return<9> ret;
90 for (int i = 0; i < 8; ++i)
91 {
92 const uint8_t v = (rgba >> (i * 4)) & 0xF;
93 ret.text[i] = v < 10 ? ('0' + v) : ('A' + v - 10);
94 }
95 return ret;
96 }
