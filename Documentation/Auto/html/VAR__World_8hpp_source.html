19 // The VWorld will manage In-Game Actors and Objects at Runtime
20 // It will automatically detect different types of Actors and
21 // will exectute their Poll every Frame
22 class VWorld {
23 public:
24 VWorld() = default;
25
26 template <typename T, typename... Args>
27 std::shared_ptr<T> CreateActor(Args &&...args)
28 {
29 static_assert(std::is_base_of_v<AActor, T>, "T must inherit from AActor");
30 auto entity = std::make_shared<T>(std::forward<Args>(args)...);
31
32 // Reuse ID if available, otherwise increment
33 VActorID id;
34 if (!m_FreeIDs.empty())
35 {
36 id = m_FreeIDs.front();
37 m_FreeIDs.pop();
38 }
39 else
40 {
41 id = m_NextID++;
42 }
43
44 entity->id = id;
45 m_Actors[id] = entity;
46 return entity;
47 }
48
49 std::shared_ptr<AActor> GetActor(VActorID id)
50 {
51 auto it = m_Actors.find(id);
52 if (it != m_Actors.end())
53 {
54 return it->second;
55 }
56 return nullptr;
57 }
58
59 const std::unordered_map<VActorID, std::shared_ptr<AActor>> &GetAllActors() { return m_Actors; }
60
61 VE::Internal::Core::Container::TVector<std::shared_ptr<AActor>> GetAllActorsList()
62 {
63 VE::Internal::Core::Container::TVector<std::shared_ptr<AActor>> list;
64 list.reserve(m_Actors.size());
65 for (const auto &[id, entity] : m_Actors)
66 {
67 list.push_back(entity);
68 }
69 return list;
70 }
71
72 void DestroyActor(VActorID id)
73 {
74 auto it = m_Actors.find(id);
75 if (it != m_Actors.end())
76 {
77 m_Actors.erase(it);
78 m_FreeIDs.push(id); // mark ID for reuse
79 }
80 }
81
82 private:
83 std::unordered_map<VActorID, std::shared_ptr<AActor>> m_Actors;
84 VActorID m_NextID = 0;
85
86 // pool of reusable IDs
87 std::queue<VActorID> m_FreeIDs;
88 };
