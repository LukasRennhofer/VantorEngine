37 // Multiply two matrices
38 VMat4 operator*(const VMat4 &rhs) const noexcept
39 {
40 VMat4 result{};
41 for (int row = 0; row < 4; ++row)
42 {
43 for (int col = 0; col < 4; ++col)
44 {
45 float sum = 0.f;
46 for (int i = 0; i < 4; ++i)
47 {
48 sum += m[row * 4 + i] * rhs.m[i * 4 + col];
49 }
50 result.m[row * 4 + col] = sum;
51 }
52 }
53 return result;
54 }

99 // Create lookAt matrix (right-handed)
100 static VMat4 LookAt(const VVector3 &eye, const VVector3 &center, const VVector3 &up) noexcept
101 {
102 VVector3 f = (center - eye).Normalized();
103 VVector3 s = f.Cross(up).Normalized();
104 VVector3 u = s.Cross(f);
105
106 VMat4 mat = Identity();
107 mat.m[0] = s.x;
108 mat.m[1] = u.x;
109 mat.m[2] = -f.x;
110 mat.m[4] = s.y;
111 mat.m[5] = u.y;
112 mat.m[6] = -f.y;
113 mat.m[8] = s.z;
114 mat.m[9] = u.z;
115 mat.m[10] = -f.z;
116
117 mat.m[12] = -s.Dot(eye);
118 mat.m[13] = -u.Dot(eye);
119 mat.m[14] = f.Dot(eye);
120 return mat;
121 }

123 static VMat4 Perspective(float fovDegrees, float aspectRatio, float nearPlane, float farPlane) noexcept
124 {
125 float fovRad = fovDegrees * 3.14159265358979323846f / 180.f;
126 float f = 1.f / std::tan(fovRad / 2.f);
127
128 VMat4 mat{};
129 mat.m[0] = f / aspectRatio; // scale X
130 mat.m[5] = f; // scale Y
131 mat.m[10] = (farPlane + nearPlane) / (nearPlane - farPlane); // Z scaling
132 mat.m[11] = -1.f; // perspective divide
133 mat.m[14] = (2.f * farPlane * nearPlane) / (nearPlane - farPlane);
134 mat.m[15] = 0.f;
135
136 return mat;
137 }

148 static VMat4 Orthographic(float left, float right, float bottom, float top, float nearPlane, float farPlane) noexcept
149 {
150 VMat4 mat{};
151
152 float rl = right - left;
153 float tb = top - bottom;
154 float fn = farPlane - nearPlane;
155
156 assert(rl != 0.f && tb != 0.f && fn != 0.f);
157
158 mat.m[0] = 2.f / rl;
159 mat.m[5] = 2.f / tb;
160 mat.m[10] = -2.f / fn;
161
162 mat.m[12] = -(right + left) / rl;
163 mat.m[13] = -(top + bottom) / tb;
164 mat.m[14] = -(farPlane + nearPlane) / fn;
165
166 mat.m[15] = 1.f;
167
168 return mat;
169 }

171 constexpr const float *Data() const noexcept { return m.data(); }
172 float *Data() noexcept { return m.data(); }

190 // Multiply two matrices

192 {
193 VMat3 result{};
194 for (int row = 0; row < 3; ++row)
195 {
196 for (int col = 0; col < 3; ++col)
197 {
198 float sum = 0.f;
199 for (int i = 0; i < 3; ++i)
200 {
201 sum += m[row * 3 + i] * rhs.m[i * 3 + col];
202 }
203 result.m[row * 3 + col] = sum;

205 }

209 // Multiply matrix by vector
210 VVector3 operator*(const VVector3 &v) const noexcept
211 {
212 return VVector3{m[0] * v.x + m[1] * v.y + m[2] * v.z, m[3] * v.x + m[4] * v.y + m[5] * v.z, m[6] * v.x + m[7] * v.y + m[8] * v.z};
213 }

257 // Create a LookAt rotation matrix (only rotation, no translation)
258 static VMat3 LookAt(const VVector3 &eye, const VVector3 &center, const VVector3 &up) noexcept
259 {
260 VVector3 f = (center - eye).Normalized();
261 VVector3 s = f.Cross(up).Normalized();
262 VVector3 u = s.Cross(f);
263
264 VMat3 mat{};
265 // Each row is a basis vector of the new orientation
266 mat.m[0] = s.x;
267 mat.m[1] = s.y;
268 mat.m[2] = s.z;
269 mat.m[3] = u.x;
270 mat.m[4] = u.y;
271 mat.m[5] = u.z;
272 mat.m[6] = -f.x;
273 mat.m[7] = -f.y;
274 mat.m[8] = -f.z;
275
276 return mat;

297 // ----------------- VMat2 -----------------
298 struct VMat2
299 {
300
301 // 2x2 matrix stored in column-major order:
302 // [ m[0] m[2] ]
303 // [ m[1] m[3] ]
304 std::array<float, 4> m{};
305
306 // Identity matrix
307 static constexpr VMat2 Identity() noexcept
308 {
309 VMat2 mat{};
310 mat.m[0] = 1.f;
311 mat.m[3] = 1.f;
312 return mat;
313 }
314
315 // Multiply two 2x2 matrices
316 VMat2 operator*(const VMat2 &rhs) const noexcept
317 {
318 VMat2 result{};
319 result.m[0] = m[0] * rhs.m[0] + m[2] * rhs.m[1];
320 result.m[1] = m[1] * rhs.m[0] + m[3] * rhs.m[1];
321 result.m[2] = m[0] * rhs.m[2] + m[2] * rhs.m[3];
322 result.m[3] = m[1] * rhs.m[2] + m[3] * rhs.m[3];
323 return result;
324 }
325
326 // Multiply by a vector (2D)
327 VVector2 operator*(const VVector2 &vec) const noexcept { return {m[0] * vec.x + m[2] * vec.y, m[1] * vec.x + m[3] * vec.y}; }
328
329 // Transpose the matrix
330 VMat2 Transposed() const noexcept
331 {
332 VMat2 result{};
333 result.m[0] = m[0];
334 result.m[1] = m[2];
335 result.m[2] = m[1];
336 result.m[3] = m[3];
337 return result;
338 }
339
340 // Compute determinant
