27 struct VLinearAllocator
28 {
29 uint8_t *data = nullptr;
30 size_t capacity = 0;
31 size_t offset = 0;
32
33 constexpr void init(void *mem, size_t size)
34 {
35 data = (uint8_t *) mem;
36 capacity = size;
37 reset();
38 }
39 constexpr uint8_t *allocate(size_t size)
40 {
41 if (offset + size >= capacity) return nullptr;
42 uint8_t *ptr = data + offset;
43 offset += size;
44 return ptr;
45 }
46 constexpr void free(size_t size)
47 {
48 size = std::min(size, offset);
49 offset -= size;
50 }
51 constexpr void reset() { offset = 0; }
52 };

54 template <typename T, size_t block_size = 256> struct VBlockAllocator
55 {
56 struct Block
57 {
58 VE::Internal::Core::Container::TVector<uint8_t> mem;
59 };
60 VE::Internal::Core::Container::TVector<Block> blocks;
61 VE::Internal::Core::Container::TVector<T *> free_list;
62
63 template <typename... ARG> inline T *allocate(ARG &&...args)
64 {
65 if (free_list.empty())
66 {
67 free_list.reserve(block_size);
68 Block &block = blocks.emplace_back();
69 block.mem.resize(sizeof(T) * block_size);
70 T *ptr = (T *) block.mem.data();
71 for (size_t i = 0; i < block_size; ++i)
72 {
73 free_list.push_back(ptr + i);
74 }
75 }
76 T *ptr = free_list.back();
77 free_list.pop_back();
78 return new (ptr) T(std::forward<ARG>(args)...);
79 }
80 inline void free(T *ptr)
81 {
82 ptr->~T();
83 free_list.push_back(ptr);
84 }
85
86 inline bool is_empty() const { return (blocks.size() * block_size) == free_list.size(); }
87 };

89 struct VPageAllocator
90 {
91 uint32_t page_count = 0;
92 uint32_t page_size = 0;
93 struct AllocationInternal
94 {
95 std::atomic<int> refcount{0};
96 OffsetAllocator::Allocation allocation;
97 };
98 struct AllocatorInternal
99 {
100 std::mutex locker;
101 OffsetAllocator::Allocator allocator;
102 VBlockAllocator<AllocationInternal> internal_blocks;
103 bool deferred_release_enabled = false;
104 uint64_t deferred_release_frame = 0;
105 std::deque<std::pair<OffsetAllocator::Allocation, uint64_t>> deferred_release_queue;
106 };
107 std::shared_ptr<AllocatorInternal> allocator;
108
109 constexpr uint64_t total_size_in_bytes() const { return uint64_t(page_count) * uint64_t(page_size); }
110
111 constexpr uint32_t page_count_from_bytes(uint64_t sizeInBytes) const
112 {
113 return uint32_t(VE::Internal::Math::align((uint64_t) sizeInBytes, (uint64_t) page_size) / (uint64_t) page_size);
114 }
115
116 // Initializes the allocator, only after which it can be used
117 // total_size_in_bytes : the allocator will manage this number of bytes
118 // page_size : the allocation granularity in bytes, each allocation will be aligned to this
119 // deferred_release : if false, allocations are freed immediately (suitable for CPU only allocations), otherwise they are freed after a number
120 //of frames passed (which should be used for GPU allocations)
121 void init(uint64_t total_size_in_bytes, uint32_t page_size = 64u * 1024u, bool deferred_release = false)
122 {
123 this->page_size = page_size;
124 this->page_count = page_count_from_bytes(total_size_in_bytes);
125 allocator = std::make_shared<AllocatorInternal>();
126 allocator->allocator.init(page_count, std::min(page_count, OffsetAllocator::default_maxallocations));
127 allocator->deferred_release_enabled = deferred_release;
128 allocator->deferred_release_frame = 0;
129 allocator->deferred_release_queue.clear();
130 }
131 // This needs to be called every frame if deferred release is enabled:
132 void update_deferred_release(uint64_t framecount, uint32_t buffercount)
133 {
134 if (allocator == nullptr) return;
135 std::scoped_lock lck(allocator->locker);
136 allocator->deferred_release_frame = framecount;
137 while (!allocator->deferred_release_queue.empty() && allocator->deferred_release_queue.front().second + buffercount < framecount)
138 {
139 allocator->allocator.free(allocator->deferred_release_queue.front().first);
140 allocator->deferred_release_queue.pop_front();
141 }
142 }

144 struct Allocation
145 {
146 std::shared_ptr<AllocatorInternal> allocator; // the allocator is retained so that allocation can deallocate itself
147 AllocationInternal *internal_state = nullptr; // this is pointing within the allocator which is retained by shared_ptr
148 uint64_t byte_offset = ~0ull;
149
150 Allocation() { Reset(); }
151 Allocation(const Allocation &other)
152 {
153 Reset();
154 allocator = other.allocator;
155 internal_state = other.internal_state;
156 byte_offset = other.byte_offset;
157 if (internal_state != nullptr)
158 {
159 internal_state->refcount.fetch_add(1);
160 }
161 }
162 Allocation(Allocation &&other) noexcept
163 {
164 Reset();
165 allocator = std::move(other.allocator);
166 internal_state = other.internal_state;
167 byte_offset = other.byte_offset;
168 other.allocator = nullptr;
169 other.internal_state = nullptr;
170 other.byte_offset = ~0ull;
171 }
172 ~Allocation() { Reset(); }
173 void operator=(const Allocation &other)
174 {
175 Reset();
176 allocator = other.allocator;
177 internal_state = other.internal_state;
178 byte_offset = other.byte_offset;
179 if (internal_state != nullptr)
180 {
181 internal_state->refcount.fetch_add(1);
182 }
183 }
184 void operator=(Allocation &&other) noexcept
185 {
186 Reset();
187 allocator = std::move(other.allocator);
188 internal_state = other.internal_state;
189 byte_offset = other.byte_offset;
190 other.allocator = nullptr;
191 other.internal_state = nullptr;
192 other.byte_offset = ~0ull;
193 }
194 void Reset()
195 {
196 if (IsValid() && (internal_state->refcount.fetch_sub(1) <= 1))
197 {
198 std::scoped_lock lck(allocator->locker);
199 if (allocator->deferred_release_enabled)
200 {
201 // can only be reclaimed after buffering amount of frames passed, this is usually used for GPU resources:
202 allocator->deferred_release_queue.push_back(std::make_pair(internal_state->allocation, allocator->deferred_release_frame));
203 }
204 else
205 {
206 // reclaimed immediately:
207 allocator->allocator.free(internal_state->allocation);
208 }
209 allocator->internal_blocks.free(internal_state);
210 }
211 allocator = {};
212 internal_state = nullptr;
213 byte_offset = ~0ull;
214 }
215
216 constexpr bool IsValid() const { return internal_state != nullptr; }
217 };
218
219 inline Allocation allocate(size_t sizeInBytes)
220 {
221 const uint32_t pages = page_count_from_bytes(sizeInBytes);
222 std::scoped_lock lck(allocator->locker);
223 OffsetAllocator::Allocation offsetallocation = allocator->allocator.allocate(pages);
224 Allocation alloc;
225 if (offsetallocation.offset != OffsetAllocator::Allocation::NO_SPACE)
226 {
227 alloc.allocator = allocator;
228 alloc.internal_state = allocator->internal_blocks.allocate();
229 alloc.internal_state->refcount.store(1);
230 alloc.internal_state->allocation = offsetallocation;
231 alloc.byte_offset = offsetallocation.offset * page_size;
232 }
233 return alloc;
234 }
235
236 inline bool is_empty() { return allocator->allocator.storageReport().totalFreeSpace == page_count; }
237 };
