25 // Custom TVector container template class
26 // T: element type, A: allocator type (defaults to std::allocator<T>)
27 template <typename T, typename A = std::allocator<T>> class TVector
28 {
29 public:
30 using value_type = T;
31
32 // Default constructor - creates empty TVector or with specified initial size
33 inline TVector(size_t initial_size = 0) { resize(initial_size); }
34
35 // Copy constructor - creates deep copy of another TVector
36 inline TVector(const TVector<T> &source_TVector) { copy_from(source_TVector); }

41 // Initializer list constructor - creates TVector from brace-enclosed list
42 inline TVector(std::initializer_list<T> init_list)
43 {
44 reserve(init_list.size());
45 for (auto &element : init_list)

48 }

51 // Destructor - cleans up all elements and deallocates memory
52 inline ~TVector()
53 {
54 clear();
55 if (m_data_ptr != nullptr)

57 m_allocator.deallocate(m_data_ptr, m_capacity);
58 }
59 }

75 // Array subscript operator - provides direct access to element at index (no bounds checking)
76 constexpr T &operator[](size_t element_index) noexcept
77 {

89 // Reserve memory for at least the specified number of elements (doesn't change size)
90 inline void reserve(size_t requested_capacity)
91 {
92 if (requested_capacity > m_capacity)
93 {
94 size_t old_capacity = m_capacity;
95 T *old_data_ptr = allocate_and_move(requested_capacity);
96 if (old_data_ptr != nullptr)
97 {
98 m_allocator.deallocate(old_data_ptr, old_capacity);
99 }
100 }
101 }
102
103 // Resize TVector to contain exactly the specified number of elements
104 inline void resize(size_t new_size)
105 {
106 if (new_size > m_current_size)
107 {
108 reserve(new_size);
109
110 // Expand the TVector - default-construct new elements
111 for (size_t i = m_current_size; i < new_size; ++i)
112 {
113 new (m_data_ptr + i) T();
114 }
115 m_current_size = new_size;
116 }
117 else if (new_size < m_current_size)
118 {
119 // Shrink the TVector - destroy excess elements
120 while (m_current_size > new_size)
121 {
122 pop_back();
123 }
124 }
125 }
126
127 // Reduce capacity to match current size (frees unused memory)
128 inline void shrink_to_fit()
129 {
130 if (m_capacity > m_current_size)
131 {
132 size_t old_capacity = m_capacity;
133 T *old_data_ptr = allocate_and_move(m_current_size);
134 if (old_data_ptr != nullptr)
135 {
136 m_allocator.deallocate(old_data_ptr, old_capacity);
137 }
138 }
139 }
140
141 // Construct element in-place at the end of the TVector using provided arguments
142 template <typename... ConstructorArgs> inline T &emplace_back(ConstructorArgs &&...constructor_args)
143 {
144 size_t old_capacity = m_capacity;
145 T *old_data_ptr = nullptr;
146 const size_t required_capacity = m_current_size + 1;
147
148 // Check if we need to grow the capacity
149 if (required_capacity > m_capacity)
150 {
151 // Double the capacity to amortize future allocations
152 old_data_ptr = allocate_and_move(required_capacity * 2);
153 }
154
155 // Construct new element at the end
156 T *new_element_ptr = new (m_data_ptr + m_current_size) T(std::forward<ConstructorArgs>(constructor_args)...);
157 m_current_size++;
158
159 // Clean up old allocation if we reallocated
160 if (old_data_ptr != nullptr)
161 {
162 m_allocator.deallocate(old_data_ptr, old_capacity);
163 }
164 return *new_element_ptr;
165 }
166
167 // Add element to the end of the TVector (copy version)
168 inline void push_back(const T &element) { emplace_back(element); }
169
170 // Add element to the end of the TVector (move version)
171 inline void push_back(T &&element) { emplace_back(std::move(element)); }
172
173 // Remove all elements from the TVector (calls destructors but doesn't deallocate memory)
174 constexpr void clear() noexcept
175 {
176 for (size_t i = 0; i < m_current_size; ++i)
177 {
178 m_data_ptr[i].~T();
179 }
180 m_current_size = 0;
181 }
182
183 // Remove the last element from the TVector
184 constexpr void pop_back() noexcept
185 {
186 assert(m_current_size > 0);
187 m_data_ptr[--m_current_size].~T();
188 }
189
190 // Remove element at specified position (shifts remaining elements left)
191 constexpr void erase(T *position) noexcept
192 {
193 assert(position >= begin());
194 assert(position < end());
195 std::move(position + 1, end(), position); // Shift elements left to fill gap
196 pop_back(); // Destroy the now-duplicate last element
197 }
198
199 // Get reference to the last element
200 constexpr T &back() noexcept
201 {
202 assert(m_current_size > 0);
203 return m_data_ptr[m_current_size - 1];
204 }
205
206 // Get const reference to the last element
207 constexpr const T &back() const noexcept
208 {
209 assert(m_current_size > 0);
210 return m_data_ptr[m_current_size - 1];
211 }
212
213 // Get reference to the first element
214 constexpr T &front() noexcept

216 assert(m_current_size > 0);
217 return m_data_ptr[0];
218 }
219
220 // Get const reference to the first element
221 constexpr const T &front() const noexcept
222 {
223 assert(m_current_size > 0);
224 return m_data_ptr[0];
225 }
226
227 // Get direct pointer to underlying array
228 constexpr T *data() noexcept { return m_data_ptr; }
229
230 // Get const pointer to underlying array
231 constexpr const T *data() const noexcept { return m_data_ptr; }
232
233 // Bounds-checked element access (same as operator[] but with explicit assertion)
234 constexpr T &at(size_t element_index) noexcept
235 {
236 assert(m_current_size > 0 && element_index < m_current_size);
237 return m_data_ptr[element_index];
238 }
239
240 // Const bounds-checked element access
241 constexpr const T &at(size_t element_index) const noexcept
242 {
243 assert(m_current_size > 0 && element_index < m_current_size);
244 return m_data_ptr[element_index];
245 }

247 // Get the number of elements currently in the TVector
248 constexpr size_t size() const noexcept { return m_current_size; }
249
250 // Check if the TVector contains no elements
251 constexpr bool empty() const noexcept { return m_current_size == 0; }
252
253 // Get iterator to first element
254 constexpr T *begin() noexcept { return m_data_ptr; }
255

257 constexpr const T *begin() const noexcept { return m_data_ptr; }
258
259 // Get iterator to one past the last element
260 constexpr T *end() noexcept { return m_data_ptr + m_current_size; }
261

263 constexpr const T *end() const noexcept { return m_data_ptr + m_current_size; }

265 private:
266 // Allocate new memory and move existing elements to it
267 // Returns pointer to old data if reallocation occurred, nullptr otherwise
268 inline T *allocate_and_move(size_t new_capacity)
269 {
270 if (new_capacity > m_current_size)
271 {
272 m_capacity = new_capacity;
273
274 // Allocate new memory block
275 T *new_allocation = m_allocator.allocate(m_capacity);
276
277 // Move existing elements to new location
278 for (size_t i = 0; i < m_current_size; ++i)
279 {
280 new (new_allocation + i) T(std::move(m_data_ptr[i]));
281 }
282
283 // Swap data pointers and return old allocation for cleanup
284 std::swap(m_data_ptr, new_allocation);
285 return new_allocation;
286 }
287 return nullptr;
288 }
289
290 // Copy all elements from another TVector (used by copy constructor and assignment)
291 inline void copy_from(const TVector<T> &source_TVector)
292 {
293 resize(source_TVector.size());
294 for (size_t i = 0; i < m_current_size; ++i)
295 {
296 m_data_ptr[i] = source_TVector.m_data_ptr[i];
297 }
298 }
299
300 // Transfer ownership from another TVector (used by move constructor and assignment)
301 inline void move_from(TVector<T> &&source_TVector)
302 {
303 clear();
304 if (m_data_ptr != nullptr)
305 {
306 m_allocator.deallocate(m_data_ptr, m_capacity);
307 }
308
309 // Take ownership of source TVector's data
310 m_capacity = source_TVector.m_capacity;
311 m_current_size = source_TVector.m_current_size;
312 m_data_ptr = source_TVector.m_data_ptr;
313
314 // Leave source TVector in valid empty state
315 source_TVector.m_data_ptr = nullptr;
316 source_TVector.m_current_size = 0;
317 source_TVector.m_capacity = 0;
318 }
319
320 // Members
321 T *m_data_ptr = nullptr;
322 size_t m_current_size = 0;
323 size_t m_capacity = 0;
324 A m_allocator;
325 };
