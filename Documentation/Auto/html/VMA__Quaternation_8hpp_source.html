27 // Identity quaternion

30 // Constructor from axis-angle (degrees)
31 static VQuaternion FromAxisAngle(const VVector3 &axis, float degrees) noexcept
32 {
33 float rad = degrees * VPI / 180.f;
34 float half = rad * 0.5f;
35 float sinH = std::sin(half);
36 VVector3 n = axis.Normalized();
37 return VQuaternion{n.x * sinH, n.y * sinH, n.z * sinH, std::cos(half)};
38 }

40 // Normalize
41 VQuaternion Normalized() const noexcept
42 {
43 float mag = std::sqrt(x * x + y * y + z * z + w * w);
44 return VQuaternion{x / mag, y / mag, z / mag, w / mag};
45 }

47 // Conjugate
48 VQuaternion Conjugated() const noexcept { return VQuaternion{-x, -y, -z, w}; }
49
50 // Inverse
51 VQuaternion Inverted() const noexcept
52 {
53 float normSq = x * x + y * y + z * z + w * w;
54 return Conjugated() * (1.0f / normSq);
55 }

57 // Quaternion multiplication
58 VQuaternion operator*(const VQuaternion &rhs) const noexcept
59 {
60 return VQuaternion{w * rhs.x + x * rhs.w + y * rhs.z - z * rhs.y, w * rhs.y - x * rhs.z + y * rhs.w + z * rhs.x,
61 w * rhs.z + x * rhs.y - y * rhs.x + z * rhs.w, w * rhs.w - x * rhs.x - y * rhs.y - z * rhs.z};
62 }

67 // Apply quaternion rotation to a vector
68 VVector3 Rotate(const VVector3 &v) const noexcept
69 {
70 VQuaternion qv{v.x, v.y, v.z, 0};
71 VQuaternion result = (*this) * qv * Inverted();
72 return {result.x, result.y, result.z};
73 }

75 // Convert to 3x3 matrix
76 VMat3 ToMat3() const noexcept
77 {
78 float xx = x * x, yy = y * y, zz = z * z;
79 float xy = x * y, xz = x * z, yz = y * z;
80 float wx = w * x, wy = w * y, wz = w * z;
81
82 VMat3 mat;
83 mat.m[0] = 1 - 2 * (yy + zz);
84 mat.m[1] = 2 * (xy - wz);
85 mat.m[2] = 2 * (xz + wy);
86
87 mat.m[3] = 2 * (xy + wz);
88 mat.m[4] = 1 - 2 * (xx + zz);
89 mat.m[5] = 2 * (yz - wx);
90
91 mat.m[6] = 2 * (xz - wy);
92 mat.m[7] = 2 * (yz + wx);
93 mat.m[8] = 1 - 2 * (xx + yy);
94 return mat;
95 }

118 // Construct from Euler angles (in degrees): Vpitch (X), yaw (Y), roll (Z)
119 static VQuaternion FromEulerAngles(const VVector3 &eulerDegrees) noexcept
120 {
121 float Vpitch = eulerDegrees.x * VPI / 180.0f;
122 float yaw = eulerDegrees.y * VPI / 180.0f;
123 float roll = eulerDegrees.z * VPI / 180.0f;
124
125 float cy = std::cos(yaw * 0.5f);
126 float sy = std::sin(yaw * 0.5f);
127 float cp = std::cos(Vpitch * 0.5f);
128 float sp = std::sin(Vpitch * 0.5f);
129 float cr = std::cos(roll * 0.5f);
130 float sr = std::sin(roll * 0.5f);
131
132 VQuaternion q(0, 0, 0, 0);
133 q.w = cr * cp * cy + sr * sp * sy;
134 q.x = sr * cp * cy - cr * sp * sy;
135 q.y = cr * sp * cy + sr * cp * sy;
136 q.z = cr * cp * sy - sr * sp * cy;
137 return q;
138 }

140 // Spherical linear interpolation
141 VQuaternion Slerp(const VQuaternion& other, float t) const noexcept
142 {
143 // Compute the cosine of the angle between the two vectors
144 float dot = x * other.x + y * other.y + z * other.z + w * other.w;
145
146 // If the dot product is negative, slerp won't take the shorter path
147 // So negate one quaternion to correct this
148 VQuaternion q2 = other;
149 if (dot < 0.0f) {
150 q2 = VQuaternion(-other.x, -other.y, -other.z, -other.w);
151 dot = -dot;
152 }
153
154 // If the quaternions are very close, use linear interpolation
155 if (dot > 0.9995f) {
156 VQuaternion result(
157 x + t * (q2.x - x),
158 y + t * (q2.y - y),
159 z + t * (q2.z - z),
160 w + t * (q2.w - w)
161 );
162 return result.Normalized();
163 }
164
165 // Calculate the angle between the quaternions
166 float theta_0 = std::acos(std::abs(dot));
167 float theta = theta_0 * t;
168 float sin_theta = std::sin(theta);
169 float sin_theta_0 = std::sin(theta_0);
170
171 float s0 = std::cos(theta) - dot * sin_theta / sin_theta_0;
172 float s1 = sin_theta / sin_theta_0;
173
174 return VQuaternion(
175 s0 * x + s1 * q2.x,
176 s0 * y + s1 * q2.y,
177 s0 * z + s1 * q2.z,
178 s0 * w + s1 * q2.w
179 );
180 }

182 };
