38 void assign(const char *str)
39 {
40 size_t len = std::strlen(str);
41 if (len <= SSO_THRESHOLD)
42 {
43 std::memcpy(m_sso_data, str, len + 1);
44 m_size = len;
45 m_using_sso = true;
46 }

53 m_using_sso = false;
54 }

66 TSafeString substr(size_t pos, size_t len) const
67 {
68 assert(pos + len <= m_size);
69 char temp[SSO_THRESHOLD + 1] = {};
70 std::memcpy(temp, c_str() + pos, len);
71 temp[len] = '\0';

75 u_int32_t hash() const noexcept
76 {
77 constexpr u_int32_t FNV_offset_basis = 2166136261u;
78 constexpr u_int32_t FNV_prime = 16777619u;
79
80 u_int32_t hash = FNV_offset_basis;
81 const char* ptr = c_str();
82 for (size_t i = 0; i < m_size; ++i)
83 {
84 hash ^= static_cast<u_int8_t>(ptr[i]);
85 hash *= FNV_prime;
86 }
87 return hash;

90 void replace(size_t pos, size_t len, const char *str)
91 {
92 assert(pos + len <= m_size);
93 TSafeString temp = substr(0, pos);
94 temp += str;

99 TSafeString &operator+=(const char *str)

104 std::strcat(m_sso_data, str);

107 else
108 {

112 if (!m_using_sso) m_allocator.deallocate(m_data, m_capacity + 1);
113 m_data = new_data;
114 m_capacity = new_len;
115 m_size = new_len;
116 m_using_sso = false;
117 }
118 return *this;
119 }
