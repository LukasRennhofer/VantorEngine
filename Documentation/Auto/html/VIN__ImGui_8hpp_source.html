163 inline void BuildUpDocking()
164 {
165 static bool dockspaceOpen = true;
166 static bool opt_fullscreen_persistant = true;
167 bool opt_fullscreen = opt_fullscreen_persistant;
168 static ImGuiDockNodeFlags dockspace_flags = ImGuiDockNodeFlags_PassthruCentralNode;
169
170 ImGuiWindowFlags window_flags = ImGuiWindowFlags_MenuBar | ImGuiWindowFlags_NoDocking;
171
172 if (opt_fullscreen)
173 {
174 ImGuiViewport *viewport = ImGui::GetMainViewport();
175 ImGui::SetNextWindowPos(viewport->Pos);
176 ImGui::SetNextWindowSize(viewport->Size);
177 ImGui::SetNextWindowViewport(viewport->ID);
178 ImGui::PushStyleVar(ImGuiStyleVar_WindowRounding, 0.0f);
179 ImGui::PushStyleVar(ImGuiStyleVar_WindowBorderSize, 0.0f);
180 window_flags |= ImGuiWindowFlags_NoTitleBar | ImGuiWindowFlags_NoCollapse | ImGuiWindowFlags_NoResize | ImGuiWindowFlags_NoMove
181 | ImGuiWindowFlags_NoBringToFrontOnFocus | ImGuiWindowFlags_NoNavFocus;
182 }
183
184 if (dockspace_flags & ImGuiDockNodeFlags_PassthruCentralNode) window_flags |= ImGuiWindowFlags_NoBackground;
185
186 ImGui::Begin("Vantor Dockspace", &dockspaceOpen, window_flags);
187
188 if (opt_fullscreen) ImGui::PopStyleVar(2);
189
190 ImGuiIO &io = ImGui::GetIO();
191 ImGuiStyle &style = ImGui::GetStyle();
192 float minWinSizeX = style.WindowMinSize.x;
193 style.WindowMinSize.x = 370.0f;
194
195 if (io.ConfigFlags & ImGuiConfigFlags_DockingEnable)
196 {
197 ImGuiID dockspace_id = ImGui::GetID("MyDockSpace");
198 ImGui::DockSpace(dockspace_id, ImVec2(0.0f, 0.0f), dockspace_flags);
199 }
200
201 ImGui::End();
202
203 style.WindowMinSize.x = minWinSizeX;
204 }
