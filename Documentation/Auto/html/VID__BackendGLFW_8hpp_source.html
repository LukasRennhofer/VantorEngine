35 void Update() override
36 {
37 previousKeys = currentKeys;
38
39 // --- Mouse ---
40 prevMouseX = mouseX; prevMouseY = mouseY;
41 glfwGetCursorPos(nativeWindow, &mouseX, &mouseY);
42
43 scrollDeltaX = scrollX; scrollDeltaY = scrollY;
44 scrollX = 0.0; scrollY = 0.0;
45
46 // --- Keyboard ---
47 for (auto [key, glfwKey] : keyMapping)
48 currentKeys[static_cast<int>(key)] = glfwGetKey(nativeWindow, glfwKey) == GLFW_PRESS;
49
50 // --- Mouse Buttons ---
51 for (auto [key, glfwButton] : mouseButtonMapping)
52 currentKeys[static_cast<int>(key)] = glfwGetMouseButton(nativeWindow, glfwButton) == GLFW_PRESS;
53
54 // TODO: Gamepad support (first controller)
55 }
56
57 bool IsPressed(const VInputButton& button) const override
58 {
59 if (button.isAxis) return false;
60 auto it = currentKeys.find(button.id);
61 return it != currentKeys.end() && it->second;
62 }
63
64 bool WasPressed(const VInputButton& button) const override
65 {
66 if (button.isAxis) return false;
67 auto itCurr = currentKeys.find(button.id);
68 auto itPrev = previousKeys.find(button.id);
69 return itCurr != currentKeys.end() && itPrev != previousKeys.end() && itCurr->second && !itPrev->second;
70 }
71
72 float GetAxis(const VInputButton& axis) const override
73 {
74 if (!axis.isAxis) return 0.f;
75 auto it = currentAxes.find(axis.axisId);
76 return it != currentAxes.end() ? it->second : 0.f;
77 }
78
79 void SetMouseCursorState(bool visible) override
80 {
81 if (m_CursorState != visible)
82 {
83 m_CursorState = visible;
84 glfwSetInputMode(nativeWindow, GLFW_CURSOR, visible ? GLFW_CURSOR_NORMAL : GLFW_CURSOR_DISABLED);
85 }
86 }
87
88 bool GetMouseCursorState() const override { return m_CursorState; }
89
90 VE::Math::VVector2 GetMousePosition() const override { return {float(mouseX), float(mouseY)}; }
91 VE::Math::VVector2 GetMouseDelta() const override { return {float(mouseX - prevMouseX), float(mouseY - prevMouseY)}; }
92 VE::Math::VVector2 GetScrollOffset() const { return {float(scrollX), float(scrollY)}; }
93 std::string GetDeviceName() const override { return "GLFWInputDevice"; }

158 void SetupCallbacks()
159 {
160 glfwSetWindowUserPointer(nativeWindow, this);
161
162 glfwSetScrollCallback(nativeWindow, [](GLFWwindow* wnd, double xoffset, double yoffset)
163 {
164 auto* input = static_cast<VGLFWInputDevice*>(glfwGetWindowUserPointer(wnd));
165 if (input)
166 {
167 input->scrollX += xoffset;
168 input->scrollY += yoffset;
169 }
170 });
