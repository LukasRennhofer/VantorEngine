71 // === Transform Matrix ===
72 VE::Math::VMat4 GetTransform() const
73 {
74 // 1. Scale the object in local space
75 // 2. Rotate the object around its local center
76 // 3. Then translate to world position
77
78 VE::Math::VMat4 S = VE::Math::VMat4::Scale(m_scale);
79 VE::Math::VMat4 R = m_rotation.ToMat4();
80
81 // First apply scale then rotation in local space
82 VE::Math::VMat4 RS = R * S;
83
84 // Then manually set the translation component without affecting the rotation
85 VE::Math::VMat4 result = RS;
86 result.m[12] = m_position.x; // Set translation X
87 result.m[13] = m_position.y; // Set translation Y
88 result.m[14] = m_position.z; // Set translation Z
89
90 return result;
91 }
92
93 private:
94 VE::Math::VVector3 m_position;
95 VE::Math::VQuaternion m_rotation;
96 VE::Math::VVector3 m_scale;
97
98 // We set the center of the mesh to a cube center for now
99 VE::Math::VVector3 m_pivot = {0.0f, 0.0f, 0.0f};
100 };
